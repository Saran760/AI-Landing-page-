function FAQItem({ faq, isOpen, onToggle }) {
  return (
    <div className={`kr-faq-item ${isOpen ? "kr-faq-open" : ""}`}>
      <button className="kr-faq-question" onClick={onToggle}>
        <span>{faq.q}</span>

      </button>
      {isOpen && <p className="kr-faq-answer">{faq.a}</p>}
    </div>

}

function FAQSection() {
  const [openIndex, setOpenIndex] = useState(0);
  return (
    <section id="faq" className="kr-section">
      <h2 className="kr-section-title">Frequently Asked Questions</h2>
      <div className="kr-faq-list">
        {faqs.map((faq, i) => (
          <FAQItem
            key={i}
            faq={faq}
            isOpen={openIndex === i}
            onToggle={() => setOpenIndex(openIndex === i ? -1 : i)}
          />
        ))}
      </div>
    </section>
  );
}

function RegistrationForm() {
  const [form, setForm] = useState({ name: "", email: "", phone: "" });
  const [status, setStatus] = useState("idle"); // idle | submitting | success | error

  const handleChange = (e) => {
    setForm({ ...form, [e.target.name]: e.target.value });
  };

  const handleSubmit = async (e) => {
    e.preventDefault();
    setStatus("submitting");
    try {
      // Calls the Express backend: POST /api/enquiry
      const res = await fetch("/api/enquiry", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          ...form,
          workshop: "AI & Robotics Summer Workshop",
        }),
      });
      if (!res.ok) throw new Error("Request failed");
      setStatus("success");
      setForm({ name: "", email: "", phone: "" });
    } catch (err) {
      setStatus("error");
    }
  };

  return (
    <section id="register" className="kr-section kr-section-alt">
      <h2 className="kr-section-title">Register Your Spot</h2>
      <p className="kr-section-sub">
        Fill in your details and our team will reach out with the next steps.
      </p>
      <form className="kr-form" onSubmit={handleSubmit}>
        <label className="kr-field">
          <span>Name</span>
          <input
            type="text"
            name="name"
            required
            value={form.name}
            onChange={handleChange}
            placeholder="Child's full name"
          />
        </label>
        <label className="kr-field">
          <span>Email</span>
          <input
            type="email"
            name="email"
            required
            value={form.email}
            onChange={handleChange}
            placeholder="parent@example.com"
          />
        </label>
        <label className="kr-field">
          <span>Phone Number</span>
          <input
            type="tel"
            name="phone"
            required
            value={form.phone}
            onChange={handleChange}
            placeholder="+91 98765 43210"
          />
        </label>
        <button type="submit" className="kr-btn-primary kr-btn-full" disabled={status === "submitting"}>
          {status === "submitting" ? "Submitting..." : "Submit Enquiry"}
        </button>
        {status === "success" && (
          <p className="kr-form-msg kr-form-success">
            Thanks! We've received your details and will be in touch soon.
          </p>
        )}
        {status === "error" && (
          <p className="kr-form-msg kr-form-error">
            Something went wrong — please try again in a moment.
          </p>
        )}
      </form>
    </section>
  );
}

}
